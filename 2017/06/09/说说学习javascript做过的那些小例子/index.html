说说学习javascript做过的那些小例子

发表于 2017-06-09 | 分类于 javascript |

1.给定一个随机数字组成的数组，选取其中的偶数，并排序， 返回新数组；

定义一个随机数组

var arr = [];
for(var i = 0; i < 50; i++) {
    arr[i] = parseInt(Math.random()*50);
}
var cl = arr.join(',');
document.write('给定一个随机数字组成的数组'+"["+cl+"]"+"<br>");

判断数组中偶数

var arr2 = [];
for(var i = 0; i < arr.length; i++) {
    if(arr[i] % 2 === 0) {
        arr2.push(arr[i]);
    }
}

给这些偶数升序排序

arr2.sort(function sortNumber(a,b) {
    return a-b;
})
document.write("选出上面数组的偶数并将其升序排序后这个新数组是"+"["+arr2+"]"+"<br>");

给这些偶数升序排序后去重

function find() {
var arr3 = [];
for(var i = 0; i < arr2.length; i++) {
    if(arr3.indexOf(arr2[i]) < 0) {
        arr3.push(arr2[i]);
    }
}
document.write("给上面的数组去重后新数组是"+"["+arr3+"]"+"<br>");
}
find(arr2);

给这些偶数降序排序

var arr4 = [];
for(var i = 0; i < arr.length; i++) {
    if(arr[i] % 2 === 0) {
    arr4.push(arr[i]);
    }
}
arr4.sort(function sortNumber(a,b) {
return b-a;
});
document.write("选出上面数组的偶数并将其降序排序后这个新数组是"+"["+arr4+"]"+"<br>");

总结：

  1. 随机数：parseInt(Math).random() —>范围是0-1的整数， parseInt(Math).random()*50—>范围是0-50的整数
  2. arr.join(‘,’) 把数组内的数用‘,’连接
  3. arr2.push(arr[i]) 遍历arr把arr的内容放到arr2中
  4. 数组升序与降序方法： arr.sort(function sortNumber(a,b)) { return a - b ;}) a-b是升序 b-a是降序
  5. 去重 arr3.indexOf(arr2[i] < 0)
